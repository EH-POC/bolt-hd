# Data Grid Table

Use table as a data grid like a spreadsheet with buit-in support for editable cells, copy/paste and more.

## Examples

### Basic Data Grid

Use `grid` variant to render Table with data-grid styling.

<Note title="Resizable columns" intent="warning">
  Column widths can be adjusted by dragging the borders between columns header.{' '}
  <br />
  However, if the <Highlight>width</Highlight> prop is set on a column, resizing
  will be disabled. To maintain resizable behavior, use <Highlight>
    minWidth
  </Highlight> or <Highlight>maxWidth</Highlight> instead.
</Note>


### Editable Data grid

Data-grid table supports editable cells with Input and Select components.

<Note title="Editable grid performance" intent="warning">
  To avoid unnecessary re-renders, please memoize all the props that are passed
  to editable cell components, especially props whose values change frequently,
  such as callbacks and objects. You can use{' '}
  <Highlight>React.useCallback()</Highlight> for functions and{' '}
  <Highlight>React.useMemo()</Highlight> for objects.
</Note>

```jsx live
                    index === row.index
                      ? { ...item, age: e.target.value }

                  )
                );

                      ? { ...item, directPersonnel: e.target.value }
                      : item
                  )
                );
              }}
            />
          ),
        },
      ],
      []
    );

    const PASTE_DATA_VALIDATION_RULES = React.useMemo(
      () => ({
        name: (value) =>
          typeof value === 'string' && value.trim().split(/\s+/).length === 2,
        age: (value) => Number.isInteger(Number(value)) && Number(value) > 18,
        maritalStatus: (value) =>
          ['Single', 'Complicated', 'In relationship'].includes(value),
        genderStatus: (value) => ['Male', 'Female'].includes(value),
        directPersonnel: (value) => ['Direct', 'Indirect'].includes(value),
      }),
      []
    );

    const applyValidationRules = React.useCallback(
      (accessor, value) => {
        if (accessor in PASTE_DATA_VALIDATION_RULES) {
          return PASTE_DATA_VALIDATION_RULES[accessor](value) ? value : '';
        }
        return value;
      },
      [PASTE_DATA_VALIDATION_RULES]
    );

    const extractDataByAccessors = React.useCallback(
      (item, accessors) =>
        accessors.reduce(
          (acc, accessor) =>
            accessor ? { ...acc, [accessor]: item[accessor] } : acc,
          {}
        ),
      []
    );

    const isCopiedSingleCell = React.useCallback((pastedRowValues) => {
      return (
        pastedRowValues?.length === 1 && pastedRowValues?.[0]?.length === 1
      );
    }, []);

    // Returns a function that fills all selected cells with a single value
    const updateDataWithPastedSingleCell = React.useCallback(
      ({ value, accessors, selectedCells }) =>
        (data) =>
          data.map((row, rowIdx) => {
            // Find all selected cells in this row
            const cellsInRow = selectedCells.filter(([r]) => r === rowIdx);
            if (cellsInRow.length === 0) return row;
            const updatedRow = { ...row };
            cellsInRow.forEach(([_, colIdx]) => {
              const accessor = accessors[colIdx];
              if (accessor) {
                updatedRow[accessor] = applyValidationRules(accessor, value);
              }
            });
            return updatedRow;
          }),
      [applyValidationRules]
    );

    // Returns a new row object with pasted values applied to the correct columns
    const updateDataWithPastedValues = React.useCallback(
      ({ rowData, pastedRowValues, startCol, accessors }) => {
        const updatedRow = { ...rowData };
        accessors.forEach((accessor, colIdx) => {
          const pastedIdx = colIdx - startCol;
          if (
            colIdx >= startCol &&
            pastedIdx < pastedRowValues.length &&
            pastedIdx >= 0 &&
            accessor
          ) {
            updatedRow[accessor] = applyValidationRules(
              accessor,
              pastedRowValues[pastedIdx]
            );
          }
        });
        return updatedRow;
      },
      [applyValidationRules]
    );

    const getAccessors = React.useCallback(
      (cols) => cols.map((col) => col.accessor || col.id),
      []
    );

    // Handles paste events for both single-cell and multi-cell pastes
    const onPasteHandler = React.useCallback(
      ({
        parsedData,
        startHighlightedPosition,
        endHighlightedPosition,
        selectedCells,
      }) => {
        const accessors = getAccessors(columns);
        const startRow = startHighlightedPosition.rowIndex;
        const endRow = endHighlightedPosition.rowIndex;
        const startCol = startHighlightedPosition.colIndex;
        const endCol = endHighlightedPosition.colIndex;
        if (isCopiedSingleCell(parsedData)) {
          const value = parsedData[0][0];
          const updateFn = updateDataWithPastedSingleCell({
            value,
            accessors,
            selectedCells,
          });
          setData(updateFn);
          return;
        }

        // Multi-cell paste
        const updateFn = (prev) =>
          prev.map((row, rowIdx) => {
            const pastedRowIdx = rowIdx - startRow;
            const pastedRowValues = parsedData[pastedRowIdx];
            if (!pastedRowValues) return row;

            return updateDataWithPastedValues({
              rowData: row,
              pastedRowValues,
              startCol,
              accessors,
            });
          });
        setData(updateFn);
      },
      [
        columns,
        isCopiedSingleCell,
        updateDataWithPastedSingleCell,
        updateDataWithPastedValues,
      ]
    );

    const onClearHandler = React.useCallback(
      ({ selectedCells }) => {
        const accessors = getAccessors(columns);
        const updateFn = (prev) =>
          prev.map((row, rowIdx) => {
            // Find all selected cells in this row
            const cellsInRow = selectedCells.filter(([r]) => r === rowIdx);
            if (cellsInRow.length === 0) return row;
            const updatedRow = { ...row };
            cellsInRow.forEach(([_, colIdx]) => {
              const accessor = accessors[colIdx];
              if (accessor) {
                updatedRow[accessor] = '';
              }
            });
            return updatedRow;
          });
        setData(updateFn);
      },
      [columns, getAccessors, setData]
    );

    const convertSpreadSheetStructureToData = (data) => {
      return data.split('\n').map((row) => row.split('\t'));
    };

    const convertDataToSpreadSheetStructure = (data, boundaries) => {
      const { bottom, left, right, top } = boundaries;

      const startRow = top;
      const endRow = bottom;
      const startCol = left;
      const endCol = right;

      const accessors = getAccessors(
        data[0] ? Object.keys(data[0]).map((key) => ({ id: key })) : []
      );
      return data
        .slice(startRow, endRow + 1)
        .map((row) =>
          accessors
            .slice(startCol, endCol + 1)
            .map((accessor) => (accessor ? row[accessor] : ''))
            .join('\t')
        )
        .join('\n');
    };

    const contextMenuConfig = {
      items: [
        {
          id: 'copy',
          text: 'Copy',
          icon: 'file-copy-outlined',
          onClick: ({ row, column, meta }) => {
            const accessors = columns.map((col) => col.accessor);
            const extractedData = data.map((item) =>
              extractDataByAccessors(item, accessors)
            );
            const boundaries = {
              top: meta.startHighlightedPosition.rowIndex,
              bottom: meta.endHighlightedPosition.rowIndex,
              left: meta.startHighlightedPosition.colIndex,
              right: meta.endHighlightedPosition.colIndex,
            };
            const tsv = convertDataToSpreadSheetStructure(
              extractedData,
              boundaries
            );
            navigator.clipboard.writeText(tsv);
          },
        },
        {
          id: 'paste',
          text: 'Paste',
          icon: 'save-outlined',
          onClick: ({ row, column, meta }) => {
            const {
              startHighlightedPosition,
              endHighlightedPosition,
              selectedCells,
            } = meta;
            navigator.clipboard.readText().then((text) => {
              const parsedData = convertSpreadSheetStructureToData(text);
              onPasteHandler({
                parsedData,
                startHighlightedPosition,
                endHighlightedPosition,
                selectedCells,
              });
            });
          },
        },
      ],
    };

    return (
      <Table
        data={data}
        columns={columns}
        variant="grid"
        selectable
        size={size}
        contextMenuConfig={contextMenuConfig}
        editableConfig={{
          onPaste: onPasteHandler,
          onClear: onClearHandler,
        }}
      />
    );
  }
  const tabs = [
    {
      id: 1,
      title: 'Default Size Table',
      panel: <HighlightEditableDataGrid />,
    },
    {
      id: 2,
      title: 'Compact Size Table',
      panel: <HighlightEditableDataGrid size="compact" />,
    },
  ];
  const [selectedTabId, setSelectedTabId] = React.useState(1);
  return (
    <Tabs
      id="highlight-editable-data-grid-tabs"
      tabs={tabs}
      selectedTabId={selectedTabId}
      onChange={setSelectedTabId}
    />
  );
}
```
